// DEFINITIONS

integer seq_file    ; // var to see if file exists 
integer seq_line    ; // captured text handler

reg [7:0] instruction_line;

// OPEN INSTRUCTIONS FILE

initial 
begin

  seq_file = $fopen("FILEPATH", "r");

    if (seq_file == 0) begin
      $display("FILE READ ERROR");
      $finish;
    end
end

// READ INSTRUCTIONS FROM FILE

always @(* )
begin

   if (!$feof(seq_file)) 
   begin

   seq_line = $fscanf(seq_file, "%b\n", instruction_line);

   $display ("Instruction: [%b]", instruction_line);  

   end

   else
   begin

   $finish;
   $fclose(seq_file);

   end

end
